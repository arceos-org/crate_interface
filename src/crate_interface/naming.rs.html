1//! Naming utilities for the crate interface.
2
3use quote::format_ident;
4use syn::{
5    parse_quote, punctuated::Punctuated, token::Comma, Error, Expr, FnArg, Ident, Pat, Signature,
6};
7
8/// Extract the argument list from the function signature to be used by the caller.
9///
10/// Returns `Err(Error)` with a compile error if any argument is not an identifier.
11///
12/// Receivers are ignored because they are already rejected by `validate_fn_signature`.
13///
14/// Returns `Ok(Punctuated<Expr, Comma>)` with the argument list.
15pub fn extract_caller_args(sig: &Signature) -> Result<Punctuated<Expr, Comma>, Error> {
16    let mut args = Punctuated::new();
17    for arg in &sig.inputs {
18        if let FnArg::Typed(t) = arg {
19            if let Pat::Ident(arg_ident) = &*t.pat {
20                args.push(parse_quote! { #arg_ident });
21            } else {
22                return Err(Error::new_spanned(
23                    &t.pat,
24                    "unexpected pattern in function argument",
25                ));
26            }
27        }
28    }
29    Ok(args)
30}

32/// Generate a unique identifier to guard against aliasing of trait names.
33pub fn alias_guard_name(trait_name: &Ident) -> Ident {
34    format_ident!("__MustNotAnAlias__{}", trait_name)
35}
36
37/// Generate a unique identifier to enforce namespace matching between
38/// `def_interface` and `impl_interface`.
39pub fn namespace_guard_name(namespace: &str) -> Ident {
40    format_ident!("__NamespaceGuard__{}", namespace)
41}
42
43/// Generate the extern function name (the symbol `def_interface` defines and
44/// `impl_interface` implements), based on the optional namespace, trait name,
45/// and function name.
46pub fn extern_fn_name(namespace: Option<&str>, trait_name: &Ident, fn_name: &Ident) -> Ident {
47    if let Some(ns) = namespace {
48        format_ident!("__{}_{}_{}", ns, trait_name, fn_name)
49    } else {
50        format_ident!("__{}_{}", trait_name, fn_name)
51    }
52}
53
54/// Generate the module name that contains the extern function declarations.
55///
56/// Namespaces are not included here because no two traits can have the same
57/// name in the same module, so the generated module name will always be unique.
58pub fn extern_fn_mod_name(trait_name: &Ident) -> Ident {
59    format_ident!("__{}_mod", trait_name)
60}
