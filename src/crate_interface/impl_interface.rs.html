1//! The implementation of the [`crate::impl_interface`] attribute macro.
2
3use proc_macro2::TokenStream;
4use quote::{format_ident, quote};
5use syn::{parse_quote, Error, ImplItem, ItemImpl, Type};
6
7use crate::args::ImplInterfaceArgs;
8use crate::naming::{alias_guard_name, extern_fn_name, extract_caller_args, namespace_guard_name};
9use crate::validator::validate_fn_signature;
10
11/// The implementation of the [`crate::impl_interface`] attribute macro.
12pub fn impl_interface(
13    mut ast: ItemImpl,
14    macro_arg: ImplInterfaceArgs,
15) -> Result<TokenStream, Error> {
16    let trait_name = if let Some((_, path, _)) = &ast.trait_ {
17        &path.segments.last().unwrap().ident
18    } else {
19        return Err(Error::new_spanned(ast, "expect a trait implementation"));
20    };
21    let impl_name = if let Type::Path(path) = &ast.self_ty.as_ref() {
22        path.path.get_ident().unwrap()
23    } else {
24        return Err(Error::new_spanned(ast, "expect a trait implementation"));
25    };
26
27    for item in &mut ast.items {
28        if let ImplItem::Fn(method) = item {
29            let (attrs, vis, sig, stmts) =
30                (&method.attrs, &method.vis, &method.sig, &method.block.stmts);
31            let fn_name = &sig.ident;
32            let extern_fn_name =
33                extern_fn_name(macro_arg.namespace.as_deref(), trait_name, fn_name).to_string();
34
35            // Validate signature: reject generic parameters and receivers
36            validate_fn_signature(sig)?;
37
38            let mut new_sig = sig.clone();
39            new_sig.ident = format_ident!("{}", extern_fn_name);
40
41            let args = extract_caller_args(sig)?;
42
43            let call_impl = quote! { #impl_name::#fn_name( #args ) };
44
45            let item: TokenStream = quote! {
46                #[inline]
47                #(#attrs)*
48                #vis
49                #sig
50                {
51                    {
52                        #[inline]
53                        #[export_name = #extern_fn_name]
54                        extern "Rust" #new_sig {
55                            #call_impl
56                        }
57                    }
58                    #(#stmts)*
59                }
60            };
61            *method = syn::parse2(item)?;
62        }
63    }
64
65    // generate alias guard to prevent aliasing of trait names
66    let alias_guard_name = alias_guard_name(trait_name);
67    let alias_guard = parse_quote!(const #alias_guard_name: () = (););
68    ast.items.push(alias_guard);
69
70    // generate namespace guard to enforce namespace matching
71    if let Some(ns) = macro_arg.namespace {
72        let ns_guard_name = namespace_guard_name(&ns);
73        let ns_guard = parse_quote!(const #ns_guard_name: () = (););
74        ast.items.push(ns_guard);
75    }
76
77    Ok(quote! { #ast })
78}
